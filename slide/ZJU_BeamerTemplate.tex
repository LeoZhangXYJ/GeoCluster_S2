\section{问题与数据}

\begin{frame}{本文不是简单复现 SAE，而是检验其迁移边界}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\linewidth}
      \begin{block}{原论文给出的起点}
        \small
        Naagar 等提出 PCA / Canonical AE / SAE + K-means 的无监督岩性填图框架，说明深层自编码器可学习低维光谱表示 \cite{naagar2024remote}。
      \end{block}
      \begin{block}{本项目追问的问题}
        \small
        在极干旱斑岩铜矿区，深层 SAE 是否仍明显优于浅层 AE？像元独立聚类能否得到连续的地质图斑？
      \end{block}
    \end{column}
    \begin{column}{0.48\linewidth}
      \small
      \begin{table}
        \centering
        \resizebox{\linewidth}{!}{%
          \begin{tabular}{>{\raggedright\arraybackslash}p{0.36\linewidth}
                          >{\raggedright\arraybackslash}p{0.56\linewidth}}
            \hline
            原论文不足 & 本文对应设计 \\
            \hline
            单一研究区验证 & 迁移到新疆土屋--延东斑岩铜矿区 \\
            深度增益未讨论 & 比较 Canonical AE 与 SAE 的真实差距 \\
            Bottleneck 固定 & 测试 $z=3,4,5$，观察类别塌缩 \\
            空间信息偏后处理 & 增加 Geo-DEC 空间上下文增强 \\
            地质解释偏定性 & 用波段比值修正 C1 判读置信度 \\
            \hline
          \end{tabular}}
      \end{table}
      \vspace{0.25em}
      \claim{核心主线：}光谱表示学习回答“能不能分”，Geo-DEC 回答“图斑能不能更连续”。
    \end{column}
  \end{columns}
\end{frame}

\section{方法设计}

\begin{frame}{两阶段流程把“光谱分离”和“空间连续”分开回答}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.58\linewidth}
      \centering
      \imgfit[\linewidth]{../report/assets/image.png}
    \end{column}
    \begin{column}{0.38\linewidth}
      \small
      \begin{block}{Stage 1：基线链条}
        Raw K-means $\rightarrow$ PCA + K-means $\rightarrow$ Canonical AE + K-means $\rightarrow$ SAE + K-means。
      \end{block}
      \begin{block}{Stage 2：空间增强}
        Geo-DEC 在 AE 思路上加入地质波段比值、3$\times$3 邻域统计和 DEC 聚类损失。
      \end{block}
      \vspace{0.4em}
      \claim{设计原则：}Geo-DEC 不作为“同级最优模型竞赛者”，而是检验空间上下文是否提升制图可读性。
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{五类方法承担不同实验角色}
  \small
  \begin{table}
    \centering
    \resizebox{\linewidth}{!}{%
      \begin{tabular}{>{\raggedright\arraybackslash}p{0.19\linewidth}
                      >{\raggedright\arraybackslash}p{0.20\linewidth}
                      >{\raggedright\arraybackslash}p{0.23\linewidth}
                      >{\raggedright\arraybackslash}p{0.28\linewidth}}
        \hline
        方法 & 特征/结构 & 主要配置 & 回答的问题 \\
        \hline
        Raw K-means & 原始 10 波段光谱 & $k=5$ & 不降维时的基础聚类质量 \\
        PCA + K-means & 线性主成分 & $z=3/5$ & 线性降维是否足够 \\
        Canonical AE & 浅层非线性表示 & $10\rightarrow16\rightarrow5\rightarrow16\rightarrow10$ & 浅层非线性是否已能捕获判别结构 \\
        SAE & 深层非线性表示 & $10\rightarrow32\rightarrow16\rightarrow z\rightarrow16\rightarrow32\rightarrow10$ & 增加深度是否带来额外收益 \\
        Geo-DEC & 地质指数 + 空间上下文 + DEC & $28\rightarrow64\rightarrow32\rightarrow5$ & 空间连续性是否改善 \\
        \hline
      \end{tabular}}
  \end{table}
  \vspace{0.45em}
  \begin{center}
    \claim{评价维度：}Silhouette、DBI、CHI 衡量聚类分离性；local agreement 与 isolated pixel ratio 衡量空间连续性。
  \end{center}
\end{frame}

\section{实验结果}

\begin{frame}{PCA 捕获了总方差，却没有显著改善聚类}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.34\linewidth}
      \small
      \begin{itemize}
        \item PC1 解释 \metric{91.95\%} 方差，前 3 个主成分累计解释 \metric{99.53\%}。
        \item 这说明 Sentinel-2 10 波段主要受整体亮度/反照率共同变化控制。
        \item 但 Raw K-means 到 PCA $z=3$ 的 CHI 仅从 123,897 提升到 127,301，约 \metric{+2.7\%}。
      \end{itemize}
      \vspace{0.4em}
      \begin{block}{结论}
        \scriptsize
        最大方差方向不等于最有利的聚类判别方向；SWIR 波段比值等低方差信息仍然重要。
      \end{block}
    \end{column}
    \begin{column}{0.62\linewidth}
      \centering
      \imgfit[\linewidth]{../results/cluster_pca/pca_explained_variance.png}
      \vspace{0.2em}

      \scriptsize PCA 方差解释分析
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{深层 SAE 没有显著拉开聚类分离性}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.42\linewidth}
      \small
      \begin{table}
        \centering
        \resizebox{\linewidth}{!}{%
          \begin{tabular}{lccc}
            \hline
            方法 & CHI $\uparrow$ & DBI $\downarrow$ & Recon. MSE $\downarrow$ \\
            \hline
            Canonical AE $z=5$ & \metric{147,239} & 0.7338 & 0.00466 \\
            SAE $z=5$ & 144,423 & \metric{0.6981} & \metric{0.00285} \\
            \hline
          \end{tabular}}
      \end{table}
      \vspace{0.4em}
      \begin{itemize}
        \item 浅层 AE 的聚类分离性与 SAE 几乎持平，CHI 还略高。
        \item SAE 的优势更多体现在重建精度和簇紧凑性。
        \item 这补充了原论文结论：\claim{深度增益具有场景依赖性}。
      \end{itemize}
    \end{column}
    \begin{column}{0.54\linewidth}
      \centering
      \imgfit[\linewidth]{../results/cluster_sae/sae_training_loss_z5.png}
      \vspace{0.2em}

      \scriptsize SAE $z=5$ 训练损失曲线；低重建误差不必然转化为全面领先的聚类指标。
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{过窄 bottleneck 会把不同地表类型压成同一类}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.31\linewidth}
      \centering
      \imgfit[0.98\linewidth]{../results/cluster_sae/sae_kmeans_k5_z3.png}
      \scriptsize $z=3,k=5$：最大类 \metric{48.9\%}，CHI=93,656
    \end{column}
    \begin{column}{0.31\linewidth}
      \centering
      \imgfit[0.98\linewidth]{../results/cluster_sae/sae_kmeans_k6_z4.png}
      \scriptsize $z=4,k=6$：Silhouette/DBI 较好，但最大类仍达 42.0\%
    \end{column}
    \begin{column}{0.31\linewidth}
      \centering
      \imgfit[0.98\linewidth]{../results/cluster_sae/sae_kmeans_k6_z5.png}
      \scriptsize \metric{$z=5,k=6$}：类别均衡性、CHI 与重建误差之间更稳妥
    \end{column}
  \end{columns}
  \vspace{0.6em}
  \begin{center}
    \small
    $z=5$ 不是所有指标绝对最优，而是避免类别塌缩并保留光谱细节的主实验折中配置。
  \end{center}
\end{frame}

\begin{frame}{综合指标显示：AE/SAE 强在分离，Geo-DEC 强在连续}
  \scriptsize
  \begin{table}
    \centering
    \resizebox{\linewidth}{!}{%
      \begin{tabular}{lcccccc}
        \hline
        Method & Dim. & $k$ & Silhouette $\uparrow$ & DBI $\downarrow$ & CHI $\uparrow$ & Agreement / Isolated \\
        \hline
        Raw K-means & 10 & 5 & 0.3709 & 0.8039 & 123,897 & 0.9029 / 0.0074 \\
        PCA + K-means & 3 & 5 & 0.3796 & 0.7866 & 127,301 & 0.9036 / 0.0073 \\
        SAE + K-means & 3 & 5 & 0.3735 & 0.8046 & 93,656 & 0.9204 / 0.0058 \\
        SAE + K-means & 4 & 6 & \metric{0.4577} & \metric{0.6706} & 135,377 & 0.8928 / 0.0093 \\
        SAE + K-means & 5 & 6 & 0.4165 & 0.6981 & 144,423 & 0.8891 / 0.0095 \\
        Canonical AE + K-means & 5 & 6 & 0.4166 & 0.7338 & \metric{147,239} & 0.8834 / 0.0099 \\
        Geo-DEC & 5 & 6 & 0.4157 & 0.8877 & 68,821 & \metric{0.9348 / 0.0057} \\
        \hline
      \end{tabular}}
  \end{table}
  \vspace{0.5em}
  \begin{itemize}
    \item AE/SAE 主配置显著优于 Raw 与 PCA，说明非线性表示学习有效。
    \item Geo-DEC 的 CHI/DBI 未超过 AE/SAE，但 local agreement 最高、isolated pixel ratio 最低。
    \item 指标应按目标解读：分离性与空间连续性之间存在 trade-off。
  \end{itemize}
\end{frame}

\begin{frame}{Geo-DEC 是空间制图增强，而不是替代 AE/SAE}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.50\linewidth}
      \begin{block}{Geo-DEC 增加了什么？}
        \small
        \begin{itemize}
          \item 输入从 10 维光谱扩展到 28 维特征。
          \item 增加 B11/B12、B12/B8A、B04/B02、NDVI 等地质相关比值。
          \item 计算 3$\times$3 邻域均值和标准差，引入局部空间上下文。
          \item 用 DEC 软分配和 KL 散度微调聚类中心。
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.46\linewidth}
      \begin{block}{实验说明了什么？}
        \small
        \begin{itemize}
          \item 空间上下文减少了椒盐噪声，图斑更连续。
          \item 但局部平滑可能稀释细微光谱判别差异。
          \item 因此 Geo-DEC 更适合定位为\claim{制图增强补充}，而非传统聚类指标最优模型。
        \end{itemize}
      \end{block}
      \vspace{0.35em}
      \begin{block}{空间连续性指标}
        \centering
        \begin{tabular}{>{\centering\arraybackslash}p{0.42\linewidth}
                        >{\centering\arraybackslash}p{0.42\linewidth}}
          {\Large \metric{0.9348}} & {\Large \metric{0.0057}} \\
          \scriptsize local agreement $\uparrow$ & \scriptsize isolated pixel ratio $\downarrow$ \\
          \scriptsize 全部方法最高 & \scriptsize 全部方法最低 \\
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{地质解释}

\begin{frame}{B11/B12 的方向修正把 C1 降级为候选线索}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.66\linewidth}
      \centering
      \imgfit[\linewidth]{../results/cluster_spectra/cluster_spectral_profiles.png}
    \end{column}
    \begin{column}{0.30\linewidth}
      \small
      \begin{block}{C1 的证据}
        \begin{itemize}
          \item 占比 \metric{18.0\%}。
          \item B11/B12=\metric{0.992}，为 6 类最低。
          \item B12/B8A=\metric{1.282}，为 6 类最高。
          \item 北部呈带状--块状，与部分假彩色褐红色区对应。
        \end{itemize}
      \end{block}
      \vspace{0.25em}
      \begin{block}{必须降置信}
        \scriptsize
        B11/B12 $<1$ 表示 B11 低于 B12，不能证明 B12 处发生吸收。因此 C1 只能作为光谱--空间候选线索，需地质图或野外验证。
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{结论}

\begin{frame}{四条认识支撑后续无标签遥感地质制图}
  \small
  \begin{block}{主要结论}
    \begin{enumerate}
      \item SAE+K-means 框架可迁移到极干旱斑岩铜矿区，非线性 AE 表示整体优于 Raw 与 PCA。
      \item 深度增益具有场景依赖性：Canonical AE 与 SAE 聚类分离性接近，SAE 的优势主要在 MSE 和 DBI。
      \item Bottleneck 维度会显著影响类别结构；$z=3$ 出现类别塌缩，$z=5$ 是本研究区更稳妥的折中。
      \item Geo-DEC 证明空间上下文能提升图斑连续性，但不能替代 AE/SAE 作为传统分离指标强基线。
    \end{enumerate}
  \end{block}
  \begin{block}{局限与展望}
    \scriptsize
    当前解译缺少地面真值验证；Geo-DEC 尚未系统消融 DEC 权重、邻域窗口和增强特征组合；未来可引入 DEM 地形校正、ASTER 热红外信息、已发表地质图一致性评价，以及 CAE/GNN 等更强的光谱--空间模型。
  \end{block}
\end{frame}
